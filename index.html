code .   Abre en VSC el directorio o carpeta

sudo mkdir #carpeta   Crea una nueva carpeta
sudo touch #archivo   Crea un nuevo archivo
sudo apt-get update   Muestra actualizaciones
sudo apt-get upgrade   actualiza
sudo apt install #aplicacion   Instala la aplicacion
clear   Limpia pantalla
sudo chown -R seco ~/CursoPreWork   Da permisos a la carpeta
rm #archivo   Borra el archivo
